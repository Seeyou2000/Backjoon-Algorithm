#include <iostream>

using namespace std;

int main() {
	long long int n = 0;
	long long int f;

	scanf("%lld", &n);
	getchar();
	scanf("%lld", &f);

	for (long long int i = n - n % 100; i < n - n % 100 + 99; i++) {
		if (i % f == 0) {
			if ((i - n < 10) && (i % 100 < 10))
				printf("%d", 0);
			printf("%d", i % 100);
			return 0;
		}
	}
	return 0;
}
